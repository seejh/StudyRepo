# 게임 서버에서 로그인 서버를 분리하는 이유

### 1. 부하 분리
게임 로직은 지속적이며 낮은 지연을 요구한다.
로그인의 경우 특정 시간대에 폭증(오픈, 점검, 퇴근 시간)하고 요청당 비용이 크다(DB 조회, 암호화)<br/>

로그인 요청이 폭증할 때 게임 서버와 로그인 서버를 분리하지 않으면 플레이 중인 유저도 영향을 받는다.

### 2. 보안
로그인 서버만 외부에 노출<br/>
게임 서버는 내부망에 숨김<br/>

인증, 암호화, 해킹 시도 차단 등 보안 로직을 로그인 서버에 집중.<br/>
게임 서버가 직접 외부에 노출되지 않아서 공격 표면이 줄어듬<br/>

#### 인증 토큰 방식
1. 클라 -> 로그인 서버: 아이디/패스워드 전송
2. 로그인 서버 -> 클라: 임시 토큰 발급
3. 클라 -> 게임 서버: 토큰으로 접속
4. 게임 서버 -> 로그인 서버: 토큰 검증
5. 검증 완료, 게임 시작

### 3. 부가적인 이유 (유지보수)
* 편의성
  * 업데이트 편의성
  * 로그인 로직 변경 시 게임 서버 재시작 없이 로그인 서버만 재시작
* 확장성
  * 로그인 트래픽 증가 시 로그인 서버만 늘리면 됨

using MySql.Data.MySqlClient;
using Microsoft.IdentityModel.Tokens;
using System.IdentityModel.Tokens.Jwt;
using System.Security.Claims;
using System.Text;
using BCrypt.Net;

public class AuthService
{
    private readonly IConfiguration _config;
    private readonly string _connStr;

    public AuthService(IConfiguration config)
    {
        _config = config;
        _connStr = config.GetConnectionString("MySQL");
    }

    public async Task<User?> GetUserAsync(string userId)
    {
        using var conn = new MySqlConnection(_connStr);
        await conn.OpenAsync();

        var cmd = new MySqlCommand(
            "SELECT id, user_id, password_hash FROM users WHERE user_id = @userId", conn);
        cmd.Parameters.AddWithValue("@userId", userId);

        using var reader = await cmd.ExecuteReaderAsync();
        if (!await reader.ReadAsync()) return null;

        return new User
        {
            Id = reader.GetInt32("id"),
            UserId = reader.GetString("user_id"),
            PasswordHash = reader.GetString("password_hash")
        };
    }

    public bool VerifyPassword(string password, string hash)
    {
        // bcrypt로 비밀번호 검증
        return BCrypt.Net.BCrypt.Verify(password, hash);
    }

    public string GenerateToken(int userId)
    {
        var secretKey = _config["Jwt:SecretKey"];
        var expiresIn = int.Parse(_config["Jwt:ExpiresInSeconds"]);

        var key = new SymmetricSecurityKey(Encoding.UTF8.GetBytes(secretKey));
        var creds = new SigningCredentials(key, SecurityAlgorithms.HmacSha256);

        var token = new JwtSecurityToken(
            claims: new[] { new Claim("userId", userId.ToString()) },
            expires: DateTime.UtcNow.AddSeconds(expiresIn),
            signingCredentials: creds
        );

        return new JwtSecurityTokenHandler().WriteToken(token);
    }
}

// AuthController.cs
using Microsoft.AspNetCore.Mvc;

[ApiController]
[Route("[controller]")]
public class AuthController : ControllerBase
{
    private readonly AuthService _authService;
    private readonly IConfiguration _config;

    public AuthController(AuthService authService, IConfiguration config)
    {
        _authService = authService;
        _config = config;
    }

    [HttpPost("login")]
    public async Task<IActionResult> Login([FromBody] LoginRequest request)
    {
        // 1. 유저 조회
        var user = await _authService.GetUserAsync(request.Id);
        if (user == null)
            return Unauthorized(new { error = "아이디가 없습니다" });

        // 2. 비밀번호 검증
        if (!_authService.VerifyPassword(request.Password, user.PasswordHash))
            return Unauthorized(new { error = "비밀번호가 틀렸습니다" });

        // 3. 토큰 발급
        var token = _authService.GenerateToken(user.Id);

        return Ok(new LoginResponse
        {
            Token = token,
            GameServerIp = _config["GameServer:Ip"],
            GameServerPort = int.Parse(_config["GameServer:Port"])
        });
    }
}

void onLoginPacket(Session* session, std::string token) {
    try {
        auto decoded = jwt::decode(token);
        auto verifier = jwt::verify()
            .allow_algorithm(jwt::algorithm::hs256{"your-256-bit-secret-key-here-make-it-long"})
            .with_claim("userId", jwt::claim(std::string{}));

        verifier.verify(decoded);

        // 만료 시간 확인
        auto exp = decoded.get_expires_at();
        if (exp < std::chrono::system_clock::now()) {
            session->disconnect();
            return;
        }

        int userId = std::stoi(decoded.get_payload_claim("userId").as_string());
        loadPlayerData(session, userId);

    } catch (const std::exception& e) {
        // 토큰 검증 실패
        session->disconnect();
    }
}

// Services/TokenService.cs
using StackExchange.Redis;

namespace LoginServer.Services
{
    public class TokenService
    {
        private readonly IDatabase _redis;
        private readonly int _expiresIn;

        public TokenService(IConfiguration config)
        {
            var redis = ConnectionMultiplexer.Connect(config.GetConnectionString("Redis"));
            _redis = redis.GetDatabase();
            _expiresIn = int.Parse(config["Token:ExpiresInSeconds"]);
        }

        // 토큰 생성 후 Redis에 저장
        public async Task<string> CreateTokenAsync(int userId)
        {
            // 랜덤 토큰 생성
            var token = Guid.NewGuid().ToString("N");  // "N": 하이픈 없는 32자리

            // Redis에 저장 (key: token, value: userId, TTL: 60초)
            await _redis.StringSetAsync(
                $"token:{token}",
                userId.ToString(),
                TimeSpan.FromSeconds(_expiresIn)
            );

            return token;
        }

        // 토큰 검증 (게임 서버에서 호출)
        public async Task<int?> ValidateTokenAsync(string token)
        {
            var value = await _redis.StringGetAsync($"token:{token}");

            if (value.IsNullOrEmpty) return null;

            // 검증 후 즉시 삭제 (1회용 토큰)
            await _redis.KeyDeleteAsync($"token:{token}");

            return int.Parse(value);
        }
    }
}

// Controllers/AuthController.cs
using Microsoft.AspNetCore.Mvc;
using LoginServer.Services;
using LoginServer.Models;

namespace LoginServer.Controllers
{
    [ApiController]
    [Route("[controller]")]
    public class AuthController : ControllerBase
    {
        private readonly AuthService _authService;
        private readonly TokenService _tokenService;
        private readonly IConfiguration _config;

        public AuthController(AuthService authService, TokenService tokenService, IConfiguration config)
        {
            _authService = authService;
            _tokenService = tokenService;
            _config = config;
        }

        // 클라이언트 로그인
        [HttpPost("login")]
        public async Task<IActionResult> Login([FromBody] LoginRequest request)
        {
            // 1. 유저 조회
            var user = await _authService.GetUserAsync(request.Id);
            if (user == null)
                return Unauthorized(new { error = "아이디가 없습니다" });

            // 2. 비밀번호 검증
            if (!_authService.VerifyPassword(request.Password, user.PasswordHash))
                return Unauthorized(new { error = "비밀번호가 틀렸습니다" });

            // 3. 토큰 생성 후 Redis에 저장
            var token = await _tokenService.CreateTokenAsync(user.Id);

            return Ok(new LoginResponse
            {
                Token = token,
                GameServerIp = _config["GameServer:Ip"],
                GameServerPort = int.Parse(_config["GameServer:Port"])
            });
        }

        // 게임 서버에서 토큰 검증 요청
        [HttpPost("verify")]
        public async Task<IActionResult> Verify([FromBody] VerifyRequest request)
        {
            var userId = await _tokenService.ValidateTokenAsync(request.Token);

            if (userId == null)
                return Unauthorized(new { error = "유효하지 않은 토큰" });

            return Ok(new { userId });
        }
    }
}
